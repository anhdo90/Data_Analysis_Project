# convert NaNs to some predefined data
def resolveNANs(dataframe):
    # Loop each column
    for columnName in dataframe.columns:
        # Change NAN depend on data type
        if dataframe[columnName].dtype == "object":
            dataframe[columnName] = dataframe[columnName].fillna("Nothing")
            
        elif dataframe[columnName].dtype == "int64":
            dataframe[columnName] = dataframe[columnName].fillna(0)
            
        elif dataframe[columnName].dtype == "float64":
            dataframe[columnName] = dataframe[columnName].fillna(0.0)
            
    return dataframe
